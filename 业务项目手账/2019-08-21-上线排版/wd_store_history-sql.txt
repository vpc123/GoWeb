CREATE TABLE `wd_store_pre_history` (
  `store_pre_history_id` bigint(20) NOT NULL AUTO_INCREMENT COMMENT '门店维度预签到统计的历史ID,唯一主键',
  `all_area_pre_ent_number` bigint(20) NOT NULL DEFAULT 0 COMMENT '门店维度下总的预签数',
  `all_pre_ent_user_number` bigint(20) NOT NULL DEFAULT 0 COMMENT '门店维度下总的经纪人人数',
  `all_pre_ent_avg_number` bigint(20) NOT NULL DEFAULT 0 COMMENT '门店维度下总的经纪人人均预签到个数',
  `area_pre_ent_number` bigint(20) NOT NULL DEFAULT 0 COMMENT '门店维度下区域总的预签到数',
  `area_pre_ent_user_number` bigint(20) NOT NULL DEFAULT 0 COMMENT '门店维度下区域总的经纪人个数',
  `area_pre_ent_avg_number` bigint(20) NOT NULL DEFAULT 0 COMMENT '门店维度下区域总的经纪人的人均预签到数',
  `area_name` varchar(100) COLLATE utf8mb4_bin DEFAULT NULL COMMENT '预报企业所在区域名称',
  `store_id` bigint(20) NOT NULL DEFAULT 0 COMMENT '门店ID',
   `store_name` varchar(64) CHARACTER SET utf8mb4 COLLATE utf8mb4_bin DEFAULT '' COMMENT '门店名称',
   `store_manager_name` varchar(64) CHARACTER SET utf8mb4 COLLATE utf8mb4_bin DEFAULT '' COMMENT '店长姓名',
   `mobile` varchar(20) CHARACTER SET utf8mb4 COLLATE utf8mb4_bin DEFAULT '' COMMENT '电话',
   `pre_ent_number` bigint(20) NOT NULL DEFAULT 0 COMMENT '门店的预签数',
   `pre_ent_user_number` bigint(20) NOT NULL DEFAULT 0 COMMENT '门店的有效经纪人人数',
   `avg_pre_ent_number` bigint(20) NOT NULL DEFAULT 0 COMMENT '门店下经纪人的平均预签到数',
  `store_rank` bigint(20) NOT NULL DEFAULT 0 COMMENT '门店的预报名数总排名',
  `pre_ent_dt` date DEFAULT NULL COMMENT '预签到历史统计时间',
  `created_by` bigint(20) NOT NULL DEFAULT '0' COMMENT '创建人 USER_ID',
  `created_tm` datetime(6) NOT NULL DEFAULT CURRENT_TIMESTAMP(6) COMMENT '创建时间，默认是 CURRENT_TIMESTAMP(6)',
  `updated_by` bigint(20) DEFAULT '0' COMMENT '修改人 USER_ID',
  `updated_tm` datetime(6) DEFAULT CURRENT_TIMESTAMP(6) ON UPDATE CURRENT_TIMESTAMP(6) COMMENT '修改时间，修改时 CURRENT_TIMESTAMP(6)',
  `is_deleted` tinyint(4) DEFAULT '0' COMMENT '记录逻辑状态，是否被删除，0 未删除， 1 已删除',
  PRIMARY KEY (`store_pre_history_id`),
  KEY `idx_pre_ent_dt` (`pre_ent_dt`),
  KEY `idx_store_id` (`store_id`)
) ENGINE=InnoDB AUTO_INCREMENT=0 DEFAULT CHARSET=utf8mb4 COMMENT='门店维度下的预报名统计数据';
